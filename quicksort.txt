// Online C++ compiler to run C++ program online
#include <iostream>
#include <vector>
using namespace std;
int part(vector<int> &arr,int st,int end){
    int idx=st-1;
    int j;
    int pvt=arr[end];
    for (j=st;j<end;j++){
        if(arr[j]<=pvt){
            idx++;
            swap(arr[idx],arr[j]);
        }
    }
    idx++;
    swap(arr[idx],arr[end]);
    return idx;
    
}
void quicks(vector<int> &arr,int st,int end){
    if(st<end){
        int pvtidx=part(arr,st,end);
        quicks(arr,st,pvtidx-1);
        quicks(arr,pvtidx+1,end);
    } 

}


int main() {
    vector<int> arr={3,5,6,1};
    quicks(arr,0,arr.size()-1);
    for(int i:arr){
        cout<<i<<"\t";
    }
    
    
   

    return 0;
}
