#include <iostream>
#include<vector>
using namespace std;
void quik(int arr[],int n){
    int i;
    for(i=0;i<n;i++){
        int current=arr[i];
        int j=i-1;
        while(j>=0&&current<arr[j]){
            arr[j+1]=arr[j];
            j=j-1;
        }
        arr[j+1]=current;
    }

    
    for(int k=0;k<n;k++){
        cout<<arr[k]<<endl;
    }
    
    
    
}

int main() {
    // Write C++ code here
    int arr[4]={4,3,2,1};
    int n =sizeof(arr)/sizeof(arr[0]);
    quik(arr,n);
    


    return 0;
}
